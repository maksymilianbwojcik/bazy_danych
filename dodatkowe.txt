3.

4.

SELECT MECZE.TEAM1,
	MECZE.TEAM2,
	PLAYER
FROM MECZE
INNER JOIN GOLE ON MECZE.ID = GOLE.MATCHID
AND GOLE.PLAYER LIKE 'Mario%'
INNER JOIN DRUZYNY ON MECZE.TEAM1 = DRUZYNY.ID
OR MECZE.TEAM2 = DRUZYNY.ID;

5.

select player, teamid, druzyny.coach, gtime from gole inner join druzyny on teamid=id where gtime<=10;

6.

7.

select distinct(player) from gole inner join mecze on id=matchid where mecze.stadium LIKE '%Warsaw%';

8.

9.

select teamname, count(gole.teamid) from druzyny inner join gole on id=teamid group by teamname, gole.teamid order by count(gole.teamid) DESC;

10.

select stadium, count(gole.matchid) from mecze inner join gole on id=matchid group by stadium order by count(gole.matchid) desc;
